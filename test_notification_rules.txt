rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write for notifications collection (temporary for testing)
    match /notifications/{notificationId} {
      allow read, create: if request.auth != null;
      allow write: if request.auth != null && 
        (resource.data.recipientUid == request.auth.uid || request.auth.uid == resource.data.recipientUid);
    }
    
    // Allow all other collections with basic auth
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
